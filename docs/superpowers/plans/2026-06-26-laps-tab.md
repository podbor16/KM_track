```js
// MM:SS для времён кругов (без часов если < 60 мин)
function fmtLapMs(ms) {
    if (!ms) return '—';
    const s = Math.floor(ms / 1000);
    const h = Math.floor(s / 3600);
    const m = Math.floor((s % 3600) / 60);
    const ss = s % 60;
    if (h > 0) return `${h}:${String(m).padStart(2,'0')}:${String(ss).padStart(2,'0')}`;
    return `${String(m).padStart(2,'0')}:${String(ss).padStart(2,'0')}`;
}

// Знаковое отклонение от среднего в формате ±MM:SS
function fmtDeltaMs(deltaMs) {
    const sign = deltaMs < 0 ? '−' : '+';
    const abs = Math.abs(deltaMs);
    const s = Math.floor(abs / 1000);
    const m = Math.floor(s / 60);
    const ss = s % 60;
    return `${sign}${String(m).padStart(2,'0')}:${String(ss).padStart(2,'0')}`;
}

// Агрегат по участникам за диапазон кругов [from, to] включительно
// Возвращает Map<participant_id, { laps: [], best, worst, avg, spread }>
function aggregateForRange(from, to) {
    const result = new Map();
    for (const lap of allLaps) {
        if (!filteredPids.has(lap.participant_id)) continue;
        if (lap.lap_number < from || lap.lap_number > to) continue;
        if (!result.has(lap.participant_id)) {
            result.set(lap.participant_id, { laps: [], best: Infinity, worst: 0, sumMs: 0 });
        }
        const agg = result.get(lap.participant_id);
        agg.laps.push(lap);
        if (lap.lap_ms > 0 && lap.lap_ms < agg.best) agg.best = lap.lap_ms;
        if (lap.lap_ms > agg.worst) agg.worst = lap.lap_ms;
        agg.sumMs += lap.lap_ms;
    }
    // Финализируем: avg, spread
    for (const [, agg] of result) {
        const validLaps = agg.laps.filter(l => l.lap_ms > 0);
        agg.best   = validLaps.length ? agg.best : 0;
        agg.avg    = validLaps.length ? Math.round(agg.sumMs / validLaps.length) : 0;
        agg.spread = agg.best && agg.worst ? agg.worst - agg.best : 0;
        agg.laps.sort((a, b) => a.lap_number - b.lap_number);
    }
    return result;
}
```

```js
function renderLapsTable() {
    if (document.getElementById('tab-laps').style.display === 'none') return;
    const from = lapsRangeFrom;
    const to   = lapsRangeTo;
    const aggMap = aggregateForRange(from, to);

    // Обновляем мета-строку под слайдером
    const lapCount = to - from + 1;
    const km = (lapCount * LAP_KM).toFixed(1);
    // Период: cumulative лучшего участника в диапазоне
    let periodMs = 0;
    let maxLapsInRange = 0;
    for (const [, agg] of aggMap) {
        if (agg.laps.length > maxLapsInRange) {
            maxLapsInRange = agg.laps.length;
            const lastLap = agg.laps[agg.laps.length - 1];
            const firstLap = agg.laps[0];
            if (lastLap && firstLap) {
                const prevCumulative = firstLap.cumulative_ms - firstLap.lap_ms;
                periodMs = lastLap.cumulative_ms - prevCumulative;
            }
        }
    }
    const periodStr = periodMs > 0 ? `· период ${fmtMs(periodMs)}` : '';
    document.getElementById('laps-slider-meta').textContent =
        `Показано ${lapCount} кругов · ${km} км ${periodStr}`;

    // Сортировка: по лучшему кругу ASC, DNF в конце
    const standingsMap = Object.fromEntries(allStandings.map(r => [r.id, r]));
    const rows = [];
    for (const [pid, agg] of aggMap) {
        const st = standingsMap[pid];
        if (!st) continue;
        rows.push({ pid, agg, st });
    }
    const isDnf = r => r.st.status === 'retired' || r.st.laps_completed === 0;
    rows.sort((a, b) => {
        const aDnf = isDnf(a), bDnf = isDnf(b);
        if (aDnf !== bDnf) return aDnf ? 1 : -1;
        if (!a.agg.best || !b.agg.best) return 0;
        return a.agg.best - b.agg.best;
    });

    // Рендер
    let rank = 0;
    const body = document.getElementById('laps-body');
    if (!rows.length) {
        body.innerHTML = '<tr><td colspan="8" style="text-align:center;color:var(--tri-muted);padding:24px">Нет данных за выбранный диапазон</td></tr>';
        return;
    }
    body.innerHTML = rows.map(({ pid, agg, st }) => {
        const dnf = isDnf({ st });
        if (!dnf) rank++;
        const rankCell = dnf ? '—' : rank;
        const nameLabel = st.gender === 'mixed' ? st.surname : `${st.surname} ${st.name}`;
        const genderTag = st.gender === 'mixed' ? 'Эст' : (st.gender === 'Мужчина' ? 'М' : 'Ж');
        const dnfBadge = dnf ? ' <span class="tri-dnf-tag">DNF</span>' : '';
        const isOpen = openLapsRows.has(pid);
        const rowClass = `tri-laps-summary-row${dnf ? ' tri-laps-dnf' : ''}`;

        const summaryRow = `
        <tr class="${rowClass}" onclick="toggleLapsRow(${pid})" style="cursor:pointer">
            <td class="tri-rank">${rankCell}</td>
            <td><span class="tri-name">${nameLabel}</span> <span class="tri-tag">${genderTag}</span>${dnfBadge}</td>
            <td>${agg.laps.length}</td>
            <td class="tri-laps-best">${agg.best ? fmtLapMs(agg.best) : '—'}</td>
            <td class="tri-laps-worst">${agg.worst ? fmtLapMs(agg.worst) : '—'}</td>
            <td>${agg.avg ? fmtLapMs(agg.avg) : '—'}</td>
            <td>${agg.spread ? '±' + fmtLapMs(agg.spread) : '—'}</td>
            <td><span class="tri-laps-expand${isOpen ? ' open' : ''}">▶</span></td>
        </tr>`;

        if (!isOpen) return summaryRow;

        // Детальные строки кругов
        const detailRows = agg.laps.map(lap => {
            const isBest = lap.lap_ms === agg.best && agg.best > 0;
            const delta = agg.avg ? lap.lap_ms - agg.avg : 0;
            const deltaClass = delta <= 0 ? 'tri-laps-delta-pos' : 'tri-laps-delta-neg';
            const deltaStr = agg.avg ? fmtDeltaMs(delta) : '—';
            const speed = lap.lap_ms > 0 ? (LAP_KM / (lap.lap_ms / 3600000)).toFixed(1) + ' км/ч' : '—';
            return `
            <tr>
                <td>${lap.lap_number}</td>
                <td class="${isBest ? 'tri-laps-best' : ''}">${fmtLapMs(lap.lap_ms)}${isBest ? ' <span class="tri-laps-star">★</span>' : ''}</td>
                <td>${speed}</td>
                <td class="${deltaClass}">${deltaStr}</td>
                <td>${fmtMs(lap.cumulative_ms)}</td>
            </tr>`;
        }).join('');

        return summaryRow + `
        <tr class="tri-laps-detail-row">
            <td colspan="8">
                <div class="tri-laps-detail-inner">
                    <table class="tri-laps-detail-table">
                        <thead>
                            <tr>
                                <th>Круг</th><th>Время</th><th>Скорость</th>
                                <th>±от среднего</th><th>Нарастающее</th>
                            </tr>
                        </thead>
                        <tbody>${detailRows}</tbody>
                    </table>
                </div>
            </td>
        </tr>`;
    }).join('');
}
```

```js
function updateLapsSliderMax() {
    if (!allLaps.length) return;
    const maxLap = Math.max(...allLaps.map(l => l.lap_number));
    const fromEl = document.getElementById('laps-from-slider');
    const toEl   = document.getElementById('laps-to-slider');
    if (!fromEl || !toEl) return;
    fromEl.max = maxLap;
    toEl.max   = maxLap;
    // Инициализируем to в максимум только при первой загрузке
    if (parseInt(toEl.value) <= 1 || parseInt(toEl.value) > maxLap) {
        toEl.value = maxLap;
        lapsRangeTo = maxLap;
        document.getElementById('laps-to-val').textContent = maxLap;
    }
    // Синхронизируем label
    const from = parseInt(fromEl.value);
    const to   = parseInt(toEl.value);
    document.getElementById('laps-range-label').textContent = `Круги ${from} — ${to}`;
}

function onLapsSlider() {
    const fromEl = document.getElementById('laps-from-slider');
    const toEl   = document.getElementById('laps-to-slider');
    let from = parseInt(fromEl.value);
    let to   = parseInt(toEl.value);
    if (to < from) { to = from; toEl.value = from; }
    lapsRangeFrom = from;
    lapsRangeTo   = to;
    document.getElementById('laps-from-val').textContent = from;
    document.getElementById('laps-to-val').textContent   = to;
    document.getElementById('laps-range-label').textContent = `Круги ${from} — ${to}`;
    renderLapsTable();
}
```
